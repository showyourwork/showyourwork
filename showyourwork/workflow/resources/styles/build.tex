% Key-value pairs used to assemble the hyperlinks for figures

\def\addvalue#1#2{\expandafter\gdef\csname syw@data@\detokenize{#1}\endcsname{#2}}
\def\usevalue#1{%
\ifcsname syw@data@#1\endcsname%
  \csname syw@data@#1\expandafter\endcsname%
\else%
  KEYNOTFOUND%
\fi%
}

% Generic: replace \LaTeX with \Latex + showyourwork in \maketitle
\let\oldmaketitle\maketitle
\def\maketitle{%
    \let\oldLaTeX\LaTeX
    \let\oldtoday\today
    \def\LaTeX{\oldLaTeX/\href{https://github.com/showyourwork/showyourwork}{\showyourwork}}
    \ifcsname gitHeader\endcsname
      \renewcommand{\today}{\oldtoday.~\gitHeader}
    \fi
    \oldmaketitle
    \let\LaTeX\oldLaTeX
    \let\today\oldtoday
}

% Hack `showkeys' to show margin links for figures
% See https://tex.stackexchange.com/a/580553
\renewcommand\showkeyslabelformat[1]{%
  \normalfont%
  \stackon[3pt]{%
    \IfEq*{\usevalue{#1_script}}{KEYNOTFOUND}{}{%
      \raisebox{0pt}[5pt][5pt]{%
        \href{\syw@url/blob/\syw@sha/\usevalue{#1_script}}{%
          \color{sywBlue}\faGithub%
        }%
      }%
    }%
  }{%
    \IfEq*{\usevalue{#1_cache}}{KEYNOTFOUND}{}{%
      \stackon[3pt]{%
        \raisebox{0pt}[5pt][5pt]{%
          \href{\usevalue{#1_cache}}{%
            \color{sywBlue}\faCloudDownload%
          }%
        }%
      }{%
        \stackon[3pt]{%
          \syw@DatasetIcon{1}%
        }{%
          \syw@DatasetIcon{2}%
        }%
      }%
    }{%
      \stackon[3pt]{%
        \IfEq*{\usevalue{#1_datasetOne}}{KEYNOTFOUND}{}{%        
          \raisebox{0pt}[5pt][5pt]{%
            \href{\usevalue{#1_datasetOne}}{%
              \color{sywBlue}\faDatabase%
            }%
          }%
        }%
      }{%
        \stackon[3pt]{%
          \IfEq*{\usevalue{#1_datasetTwo}}{KEYNOTFOUND}{}{%        
            \raisebox{0pt}[5pt][5pt]{%
              \href{\usevalue{#1_datasetTwo}}{%
                \color{sywBlue}\faDatabase%
              }%
            }%
          }%
        }{%
          \IfEq*{\usevalue{#1_datasetThree}}{KEYNOTFOUND}{}{%        
            \raisebox{0pt}[5pt][5pt]{%
              \href{\usevalue{#1_datasetThree}}{%
                \color{sywBlue}\faDatabase%
              }%
            }%
          }%
        }%
      }%
    }%
  }%
}
